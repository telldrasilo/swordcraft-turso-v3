# План разработки модуля локаций

## Обзор проекта

**Цель:** Создать систему локаций для IDLE-игры про кузнеца, где искатели отправляются в экспедиции за ресурсами.

**Философия мира:** Реалистичный, суровый, с элементами мистического фэнтези. Магия существует, но она опасна и непредсказуема. Избегать клише "избранного", "пророчеств" и "древнего зла".

### Ключевые интерфейсы

```typescript
// Location
interface Location {
  id: string;
  name: string;
  description: string;
  tier: 1 | 2 | 3 | 4;
  tags: LocationTag[];
  resources: LocationResource[];
  weather: WeatherType[];
  npcs: NPCType[];
  plotHook: string;
  dungeonHook: string;
  unlockRequirements: UnlockRequirements;
}

// LocationResource
interface LocationResource {
  materialId: string;
  baseWeight: number;        // Базовый шанс выпадения
  rarityModifier: number;    // Множитель редкости
  minQuantity: number;
  maxQuantity: number;
  seasonalModifier?: SeasonalModifier;
}

// MissionTemplate
interface MissionTemplate {
  id: string;
  locationId: string;
  name: string;
  description: string;
  type: 'hunt' | 'scout' | 'clear' | 'delivery' | 'magic' | 'rescue';
  difficulty: 'easy' | 'normal' | 'hard' | 'extreme';
  duration: number;
  rewards: MissionReward[];
  isRepeatable: boolean;
  enemyTypes?: string[];
}
```
