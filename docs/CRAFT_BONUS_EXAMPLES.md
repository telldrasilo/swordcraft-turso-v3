# Примеры расчёта бонусов к крафту

**Дата:** 2025-03-25  
**Версия:** 1.0

## Тестовый сценарий

Все примеры при **экспертизе 50%**

---

## 1. Железо (Iron)

### Свойства материала
```typescript
{
  physical: {
    hardness: 50,
    toughness: 45,
    elasticity: 25,
    thermalConductivity: 40,
    porosity: 15,
    tensileStrength: 30
  },
  chemical: {
    stability: 50,
    reactivity: 45
  },
  arcane: {
    conductivity: 15,
    stability: 60
  },
  processing: {
    workability: 70,
    refineDifficulty: 50,
    purityPotential: 40,
    defectRisk: 10,
    repairability: 100
  },
  economy: {
    rarity: 25,
    discoverability: 100
  }
}
```

### Расчёт бонусов

#### Скорость крафта
```
baseTimeMultiplier = 1.0 - (50 * 0.0015) = 0.925
materialTimeFactor =
  (70 / 100) * 0.35 -           // 0.245
  (50 / 200) * 0.20 -           // -0.050
  (50 / 100) * 0.15 +           // 0.075
  (40 / 100) * 0.30 =            // 0.120
  
materialTimeFactor = 0.245 - 0.050 - 0.075 + 0.120 = 0.240

timeMultiplier = 0.925 * (0.7 + 0.240 * 0.6) = 0.925 * 0.844 = 0.781
```
**Результат: -22% скорости (быстрее)**

#### Риск дефектов
```
baseRiskMultiplier = 1.0 - (50 * 0.005) = 0.75
materialRiskFactor =
  (50 / 100) * 0.25 -           // 0.125
  (10 / 100) * 0.25 +           // 0.025
  (30 / 200) * 0.25 +           // 0.0375
  (40 / 100) * 0.25 =            // 0.10
  
materialRiskFactor = 0.125 - 0.025 + 0.0375 + 0.10 = 0.2375

defectRiskMultiplier = 0.75 * (0.5 + 0.2375 * 1.0) = 0.75 * 0.7375 = 0.553
```
**Результат: -45% риска (надёжно)**

#### Отходы материала
```
baseWasteMultiplier = 1.0 - (50 * 0.006) = 0.7
materialWasteFactor =
  (1 - 15 / 100) * 0.30 -        // 0.255
  (100 / 100) * 0.35 -           // 0.35
  (70 / 100) * 0.25 -            // 0.175
  (30 / 200) * 0.10 =             // 0.015
  
materialWasteFactor = 0.255 - 0.35 - 0.175 - 0.015 = -0.285

materialWasteMultiplier = 0.7 * (0.4 + (-0.285) * 1.2) = 0.7 * 0.058 = 0.041
```
**Результат: -96% отходов (очень экономно)**

#### Качество
```
baseQualityBonus = 50 * 0.15 = 7.5
materialQualityBonus =
  (50 / 200) * 20 +               // 5.0
  (45 / 200) * 20 +               // 4.5
  (15 / 100) * 10 +               // 1.5
  (25 / 200) * 15 =               // 1.875
  
materialQualityBonus = 5.0 + 4.5 + 1.5 + 1.875 = 12.875

qualityBonus = 7.5 + 12.875 = 20.375 ≈ 20
```
**Результат: +20 качества**

#### Точность прогноза
```
baseAccuracy = 50 + 50 * 0.4 = 70
materialAccuracyFactor =
  (50 / 100) * 20 +               // 10
  (60 / 100) * 15 +               // 9
  (100 / 100) * 15 =              // 15
  
materialAccuracyFactor = 10 + 9 + 15 = 34

predictionAccuracy = 70 + 34 = 104% → 100%
```
**Результат: 100% точности (идеально)**

#### Разброс
```
baseVariance = 1.0 - (50 * 0.008) = 0.6
materialStability = (50 + 60) / 200 = 0.55
materialQuality = (50 + 45) / 400 = 0.2375

varianceMultiplier = 0.6 * (0.3 + 0.55 * 0.35 + 0.2375 * 0.35)
                  = 0.6 * (0.3 + 0.1925 + 0.083125)
                  = 0.6 * 0.575625
                  = 0.345
```
**Результат: 0.345 разброса (средний)**

### Итоговые бонусы для Железа при 50% экспертизы
- **Скорость:** -22%
- **Надёжность:** -45%
- **Экономия:** -96%
- **Качество:** +20
- **Точность:** 100%
- **Разброс:** 0.345

## 2. Мифрил (Mithril)

### Свойства материала
```typescript
{
  physical: {
    hardness: 88,
    toughness: 75,
    elasticity: 75,
    thermalConductivity: 90,
    porosity: 5,
    tensileStrength: 90
  },
  chemical: {
    stability: 90,
    reactivity: 10
  },
  arcane: {
    conductivity: 90,
    stability: 90
  },
  processing: {
    workability: 25,
    refineDifficulty: 90,
    purityPotential: 90,
    defectRisk: 15,
    repairability: 100
  },
  economy: {
    rarity: 95,
    discoverability: 20
  }
}
```

### Расчёт бонусов

#### Скорость крафта
```
baseTimeMultiplier = 0.925
materialTimeFactor =
  (25 / 100) * 0.35 -           // 0.0875
  (88 / 200) * 0.20 -           // -0.088
  (90 / 100) * 0.15 -           // -0.135
  (90 / 100) * 0.30 =            // 0.27
  
materialTimeFactor = 0.0875 - 0.088 - 0.135 + 0.27 = 0.1345

timeMultiplier = 0.925 * (0.7 + 0.1345 * 0.6) = 0.925 * 0.7807 = 0.722
```
**Результат: -28% скорости (быстро для легендарного материала)**

#### Риск дефектов
```
baseRiskMultiplier = 0.75
materialRiskFactor =
  (90 / 100) * 0.25 -           // 0.225
  (15 / 100) * 0.25 +           // 0.0375
  (90 / 200) * 0.25 +           // 0.1125
  (90 / 100) * 0.25 =            // 0.225
  
materialRiskFactor = 0.225 - 0.0375 + 0.1125 + 0.225 = 0.525

defectRiskMultiplier = 0.75 * (0.5 + 0.525 * 1.0) = 0.75 * 1.025 = 0.769
```
**Результат: -23% риска (надёжно)**

#### Качество
```
baseQualityBonus = 7.5
materialQualityBonus =
  (88 / 200) * 20 +               // 8.8
  (75 / 200) * 20 +               // 7.5
  (90 / 100) * 10 +               // 9.0
  (95 / 200) * 15 =               // 7.125
  
materialQualityBonus = 8.8 + 7.5 + 9.0 + 7.125 = 32.425

qualityBonus = 7.5 + 32.425 = 39.925 ≈ 40
```
**Результат: +40 качества (очень высокое)**

#### Точность прогноза
```
baseAccuracy = 70
materialAccuracyFactor =
  (90 / 100) * 20 +               // 18
  (90 / 100) * 15 +               // 13.5
  (20 / 100) * 15 =               // 3
  
materialAccuracyFactor = 18 + 13.5 + 3 = 34.5

predictionAccuracy = 70 + 34.5 = 104.5% → 100%
```
**Результат: 100% точности (идеально)**

#### Разброс
```
baseVariance = 0.6
materialStability = (90 + 90) / 200 = 0.9
materialQuality = (88 + 75) / 400 = 0.4075

varianceMultiplier = 0.6 * (0.3 + 0.9 * 0.35 + 0.4075 * 0.35)
                  = 0.6 * (0.3 + 0.315 + 0.1426)
                  = 0.6 * 0.7576
                  = 0.455
```
**Результат: 0.455 разброса (низкий)**

### Итоговые бонусы для Мифрила при 50% экспертизы
- **Скорость:** -28%
- **Надёжность:** -23%
- **Экономия:** (аналогично рассчитывается)
- **Качество:** +40
- **Точность:** 100%
- **Разброс:** 0.455
